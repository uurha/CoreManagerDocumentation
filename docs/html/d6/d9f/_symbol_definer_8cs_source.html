50  public void DefineSymbol(string key)
51  {

53  _symbols[key] = !value;
54  SetScriptingDefine(_symbols.FirstOrDefault(x => x.Key.Equals(key)));

57  private void SetScriptingDefine(KeyValuePair<string, bool> pair)
58  {
59  var allDefines = AllDefines();
60  allDefines.RemoveAll(string.IsNullOrWhiteSpace);
61 
62  if (pair.Value)
63  {
64  if (!allDefines.Contains(pair.Key)) allDefines.Add(pair.Key);
65  }
66  else
67  {
68  allDefines.RemoveAll(x => x == pair.Key);

71  PlayerSettings.SetScriptingDefineSymbolsForGroup(
72  EditorUserBuildSettings.selectedBuildTargetGroup,
73  string.Join(";", allDefines.ToArray()));

75  }
