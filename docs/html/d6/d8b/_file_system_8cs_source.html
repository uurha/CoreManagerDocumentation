135  private async void LoadInternal<T>(string fileName, Action<T> onLoaded, Action<Exception> onError, Object context)
136  where T : Unique
137  {

139  {
140  await _semaphoreSlim.WaitAsync();
141  var bufferPath = Path.Combine(_defaultPath, fileName, _defaultExtension);
142 
143  if (!(Directory.Exists(_defaultPath) && File.Exists(bufferPath)))
144  {
145  onLoaded?.Invoke(null);
146  return;
147  }
148  using var stream = new FileStream(bufferPath, FileMode.Open);
149  using var writer = new StreamReader(stream);
150  var loadedData = await writer.ReadToEndAsync();
151  onLoaded?.Invoke(JsonUtility.FromJson<T>(loadedData));
152  }
